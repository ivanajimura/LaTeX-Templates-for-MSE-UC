\begin{document}
\begin{enumerate}
    \item Use \highlight{\textbackslash begin\{enumerate\}} to create ordered lists
    \begin{enumerate}
        \item \highlight{\textbackslash begin\{enumerate\}} can also create subitems
    \end{enumerate}
\end{enumerate}

\begin{itemize}
    \item \highlight{\textbackslash begin\{itemize\}} to create unordered lists
\end{itemize}

\begin{description}
    \item [\highlight{\textbackslash begin\{description\}}] Description can be used to create items in bold followed by their description
    \item [Syntax] [Name in brackets] Description after the right bracket
\end{description}


\end{document}
